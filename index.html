window.openRandomWiki = async function(){ try { const response = await fetch( "https://en.wikipedia.org/w/api.php?action=query&list=categorymembers&cmtitle=Category:Baseball&cmnamespace=0&cmlimit=100&origin=*&format=json" ); const data = await response.json(); const pages = data.query.categorymembers; // Filter out category-like or list pages const validPages = pages.filter(p => !p.title.toLowerCase().includes("list of") && !p.title.toLowerCase().includes("outline of") && !p.title.toLowerCase().includes("index of") ); if(validPages.length === 0){ alert("No baseball articles found."); return; } const randomPage = validPages[Math.floor(Math.random()*validPages.length)]; window.open( `https://en.wikipedia.org/wiki/${encodeURIComponent(randomPage.title)}`, "_blank" ); } catch(err){ alert("Wikipedia connection failed."); } };
